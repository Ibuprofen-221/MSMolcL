# 3.2 聚合操作阻塞状态轮询请求 — 解决方案

## 问题定位

`backend/api/file_upload_status.py` 的 `GET /api/upload-files/status` 在请求处理中直接调用 `get_mgf_only_batch_status()`，后者内部同步执行 `_run_aggregate_if_needed()`：

```python
# backend/api/file_upload_status.py:12-27
@file_upload_status_router.get("/upload-files/status")
async def upload_files_status_api(task_id, current_user):
    # 这个调用内部包含 _run_aggregate_if_needed()
    data = get_mgf_only_batch_status(current_user.data_path, task_id)
    return success_response(message="上传任务状态", data=data)

# backend/services/sirius_batch_service.py:386
def get_mgf_only_batch_status(user_data_path, task_id):
    # ...
    _run_aggregate_if_needed(user_data_path, task_id, task_dir, meta)
    # 聚合可能耗时 10-60+ 秒
```

`_run_aggregate_if_needed` 调用 `preprocess_main()` 对每个成功完成的 Sirius 任务做预处理，对于多文件批次，这个操作可能耗时数十秒。

## 影响

- 前端每 2 秒轮询一次 `/api/upload-files/status`
- 当聚合被触发时，**该轮询请求会阻塞到聚合完成才返回**
- HTTP 请求可能因超时而断开（浏览器/代理默认 30-60s）
- 用户看到 "pending" 状态停留很久然后突然完成，无进度反馈

## 解决方案

### 核心思路

将「状态查询」与「聚合执行」解耦：
1. 状态查询接口**立即返回**当前状态 + 聚合进度标记
2. 聚合操作在**后台异步任务**中执行
3. 聚合完成后更新 meta 文件，下次轮询自动感知

### Step 1: 修改 `get_mgf_only_batch_status` — 移除同步聚合

```python
# backend/services/sirius_batch_service.py（修改）

def get_mgf_only_batch_status(user_data_path: str, task_id: str) -> dict:
    """仅查询状态，不触发聚合。聚合由调用方异步触发。"""
    task_dir = get_user_task_dir(user_data_path=user_data_path, task_id=task_id, create=False)
    if not task_dir.exists() or not task_dir.is_dir():
        raise SiriusBatchError("任务不存在")

    meta = _read_meta(task_dir)
    if str(meta.get("mode") or "") != "mgf_only_async":
        raise SiriusBatchError("当前任务不是 mgf-only 异步任务")

    app = _get_celery_app()
    items = [x for x in meta.get("items", []) if isinstance(x, dict)]

    # 更新 Celery 任务状态
    for item in items:
        current_status = str(item.get("status") or "pending")
        if current_status in {"success", "failed"}:
            continue
        # ... Celery 状态查询逻辑不变 ...

    meta["items"] = items
    _write_meta(task_dir, meta)

    _, progress = _calc_overall(items)

    # 判断是否需要聚合
    needs_aggregate = (
        not bool(meta.get("aggregate_done"))
        and progress["done"] >= progress["total"]
    )

    aggregate_status = meta.get("aggregate_status", "pending")

    response_data = {
        "task_id": task_id,
        "status": aggregate_status if bool(meta.get("aggregate_done")) else "processing",
        "upload_mode": "mgf_only_async",
        "progress": progress,
        "needs_aggregate": needs_aggregate,                # 新增：是否需要触发聚合
        "aggregate_done": bool(meta.get("aggregate_done")), # 新增：聚合是否完成
        "aggregate_status": aggregate_status,
        "files": meta.get("final_files") if isinstance(meta.get("final_files"), dict) else {},
        "batch_summary": meta.get("batch_summary") if isinstance(meta.get("batch_summary"), dict) else {},
        "errors": [...],
    }

    # 如果聚合已完成，附带 statas 数据
    files = response_data.get("files") or {}
    statas_path = Path(str(files.get("statas_path") or ""))
    if statas_path.is_file():
        response_data["statas"] = _read_json(statas_path)

    return response_data
```

### Step 2: 新增异步聚合触发函数

```python
# backend/services/sirius_batch_service.py（新增）

import asyncio
import logging

logger = logging.getLogger(__name__)

# 防止重复触发聚合
_aggregate_locks: dict[str, asyncio.Lock] = {}


async def ensure_aggregate_async(user_data_path: str, task_id: str) -> None:
    """确保聚合任务已触发（幂等）。在后台线程池中执行聚合。"""
    lock_key = f"{user_data_path}:{task_id}"
    lock = _aggregate_locks.setdefault(lock_key, asyncio.Lock())

    async with lock:
        # 双重检查：可能已被并发请求触发
        task_dir = get_user_task_dir(user_data_path=user_data_path, task_id=task_id, create=False)
        meta = _read_meta(task_dir)
        if bool(meta.get("aggregate_done")):
            return  # 已聚合完成

        items = [x for x in meta.get("items", []) if isinstance(x, dict)]
        _, progress = _calc_overall(items)
        if progress["done"] < progress["total"]:
            return  # 还有任务未完成

        # 标记"聚合进行中"防止重复
        meta["aggregate_status"] = "running"
        _write_meta(task_dir, meta)

        # 在线程池中执行 CPU 密集型聚合
        loop = asyncio.get_running_loop()
        try:
            await loop.run_in_executor(
                None,
                _run_aggregate_if_needed_sync,
                user_data_path, task_id, task_dir, meta,
            )
        except Exception as exc:
            logger.error("聚合失败 task=%s: %s", task_id, exc)
            meta["aggregate_done"] = True
            meta["aggregate_status"] = "failed"
            meta["aggregate_error"] = str(exc)
            _write_meta(task_dir, meta)
        finally:
            _aggregate_locks.pop(lock_key, None)
```

### Step 3: 修改状态查询 API 触发聚合

```python
# backend/api/file_upload_status.py（修改）

from services.sirius_batch_service import get_mgf_only_batch_status, ensure_aggregate_async

@file_upload_status_router.get("/upload-files/status")
async def upload_files_status_api(
    task_id: str = Query(...),
    current_user: User = Depends(get_current_user),
):
    safe_task_id = (task_id or "").strip()
    if not safe_task_id:
        raise HTTPException(status_code=400, detail="task_id 不能为空")

    # 1. 快速获取状态（不触发聚合）
    data = get_mgf_only_batch_status(
        user_data_path=current_user.data_path,
        task_id=safe_task_id,
    )

    # 2. 如果需要聚合，异步触发（不 await，让聚合在后台跑）
    if data.get("needs_aggregate"):
        asyncio.create_task(
            ensure_aggregate_async(current_user.data_path, safe_task_id)
        )
        # 立即返回，告诉前端"聚合正在进行中"
        data["status"] = "aggregating"
        data["message"] = "所有 Sirius 任务已完成，正在聚合结果..."

    return success_response(message="上传任务状态", data=data)
```

### Step 4: 调整前端轮询逻辑

前端在收到 `"aggregating"` 状态时，可以显示专门的进度文本，无需任何代码改动（因为轮询机制不变，只是后端不再阻塞）：

```javascript

if (statusData.status === 'aggregating') {
  // 显示 "正在聚合结果，请稍候..."
}
```
