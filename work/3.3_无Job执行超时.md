# 3.3 无 Job 执行超时 — 解决方案

## 问题定位

`backend/services/retrieve_runtime.py` 中的 GPU Worker 主循环对每个 job 调用 `execute_retrieve_job()` 时无超时限制：

```python
# backend/services/retrieve_runtime.py:82-106

try:
    result = execute_retrieve_job(job_payload)  # ← 无超时限制
    response_queue.put({...})
except Exception as exc:
    response_queue.put({...})
```

如果 GPU 推理过程卡死（CUDA 死锁、OOM 后挂起、模型推理无限循环），Worker 进程将永久阻塞在 `execute_retrieve_job()` 中，无法处理后续任务。

## 影响

- 该 ion_mode 的所有后续作业永远无法执行
- 无告警、无外部感知（直到有人查询状态并触发 `_sync_worker_state`）
- 问题 2.6 的健康检查虽可检测进程崩溃，但无法检测进程"存活但卡死"的情况

## 解决方案

### 核心思路

使用子进程执行单个 job，并设置 `join(timeout=N)`。超时后终止子进程并标记任务失败。

### 完整实现

```python
# backend/services/retrieve_runtime.py（修改 _gpu_worker_main）

import signal
import multiprocessing
from multiprocessing import Process, Queue as MpQueue

JOB_EXECUTION_TIMEOUT = 600  # 单个 job 最大执行时间（秒）


def _execute_job_in_subprocess(
    job_payload: dict,
    result_queue: MpQueue,
    ion_mode: str,
) -> None:
    """
    在独立子进程中执行 job，结果通过 result_queue 传回。

    与 Worker 主进程隔离开：即使 CUDA 操作卡死，
    Worker 主进程可以通过 terminate() 强制终止。
    """
    try:
        # 子进程需要重新初始化模型（因为 CUDA context 不能在进程间继承）
        from services.retrieve import init_retrieve_process
        init_retrieve_process(ion_mode=ion_mode)

        from services.retrieve_service import execute_retrieve_job
        result = execute_retrieve_job(job_payload)
        result_queue.put({"status": "success", "result": result})
    except Exception as exc:
        result_queue.put({"status": "failed", "error": str(exc)})


def _gpu_worker_main(request_queue, response_queue, ion_mode: str) -> None:
    """GPU专用进程：按离子模式常驻模型，用子进程隔离每个 job 的执行。"""
    try:
        init_retrieve_process(ion_mode=ion_mode)
    except Exception as exc:
        response_queue.put({
            "type": "worker_error",
            "ion_mode": ion_mode,
            "error": f"GPU进程模型初始化失败({ion_mode}): {exc}",
            "at": _utc_now_iso(),
        })
        return

    while True:
        try:
            message = request_queue.get(timeout=1.0)
        except Empty:
            continue

        if message is None:
            break

        job_id = message.get("job_id")
        job_payload = message.get("payload")

        if not job_id or not isinstance(job_payload, dict):
            response_queue.put({
                "type": "done", "job_id": job_id,
                "ion_mode": ion_mode, "status": "failed",
                "result": None, "error": "任务消息格式无效",
                "finished_at": _utc_now_iso(),
            })
            continue

        response_queue.put({
            "type": "running", "job_id": job_id,
            "ion_mode": ion_mode, "started_at": _utc_now_iso(),
        })

        # ── 关键修改：用子进程执行 job 并设置超时 ──
        mp_ctx = multiprocessing.get_context("spawn")
        result_queue = mp_ctx.Queue(maxsize=1)

        subprocess = mp_ctx.Process(
            target=_execute_job_in_subprocess,
            args=(job_payload, result_queue, ion_mode),
            name=f"retrieve-job-{job_id}",
        )
        subprocess.start()
        subprocess.join(timeout=JOB_EXECUTION_TIMEOUT)

        if subprocess.is_alive():
            # 超时：强制终止子进程
            subprocess.terminate()
            subprocess.join(timeout=5)
            if subprocess.is_alive():
                subprocess.kill()
                subprocess.join(timeout=2)

            response_queue.put({
                "type": "done", "job_id": job_id,
                "ion_mode": ion_mode, "status": "failed",
                "result": None,
                "error": f"任务执行超时（{JOB_EXECUTION_TIMEOUT}秒）",
                "finished_at": _utc_now_iso(),
            })
            continue

        # 正常完成：从 result_queue 读取结果
        try:
            result_msg = result_queue.get(timeout=5)
        except Empty:
            response_queue.put({
                "type": "done", "job_id": job_id,
                "ion_mode": ion_mode, "status": "failed",
                "result": None,
                "error": "任务子进程已完成但未返回结果",
                "finished_at": _utc_now_iso(),
            })
            continue

        if result_msg["status"] == "success":
            response_queue.put({
                "type": "done", "job_id": job_id,
                "ion_mode": ion_mode, "status": "success",
                "result": result_msg["result"], "error": None,
                "finished_at": _utc_now_iso(),
            })
        else:
            response_queue.put({
                "type": "done", "job_id": job_id,
                "ion_mode": ion_mode, "status": "failed",
                "result": None, "error": result_msg.get("error", "未知错误"),
                "finished_at": _utc_now_iso(),
            })
```

### 替代方案（更轻量，如果子进程开销太大）

如果 GPU 模型加载时间过长导致子进程方案不可行，可以使用 `signal.alarm` 在同一进程中设置超时（仅限 Linux）：

```python
import signal

def _timeout_handler(signum, frame):
    raise TimeoutError("任务执行超时")

# 在 execute_retrieve_job 调用前设置
signal.signal(signal.SIGALRM, _timeout_handler)
signal.alarm(JOB_EXECUTION_TIMEOUT)
try:
    result = execute_retrieve_job(job_payload)
finally:
    signal.alarm(0)  # 取消超时
```

**注意**：`signal.alarm` 仅适用于主线程，且与 `multiprocessing` 的 `spawn` 模式兼容。若 CUDA 层不响应信号，此方案无效。

### 配置化超时时间

不同检索类型可能需要不同的超时时间：

```python
# backend/core/config.py（新增）

retrieve_job_timeout_seconds = 600      # 普通检索 job 超时：10 分钟
retrieve_advanced_job_timeout_seconds = 900  # 高级检索 job 超时：15 分钟
```

### 同样修改 `retrieve_advanced_runtime.py`

`backend/services/retrieve_advanced_runtime.py` 的 `_gpu_worker_main` 需做相同改造。
