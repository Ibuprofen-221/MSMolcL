# 4.1 无用户存储配额 — 解决方案

## 问题定位

系统对用户存储没有任何配额限制：

```python
# backend/api/file_upload.py — 上传接口
# 仅限制单文件大小（10MB），无每用户总存储上限
# 仅限制单次上传文件数（50个），无任务总数上限

# backend/services/history_store.py — 删除逻辑
# _remove_task_dir_if_exists 会删除任务目录，但用户可能从不删除
```

用户可以通过以下方式无限消耗磁盘：
1. 反复上传大文件创建新 task
2. 每次上传最多 50 个文件 × 10MB = 500MB
3. 任务目录永不清除

## 影响

- 恶意用户可填满服务器磁盘
- 正常用户也可能因为忘记清理而积累大量数据
- 无任何监控或告警

## 解决方案

### 总体策略

三层防护：
1. **硬限制**：每用户最大存储容量（如 5GB）和最大任务数（如 100 个）
2. **自动清理**：任务超过 N 天后自动删除
3. **前端提示**：在接近限额时展示警告

### Step 1: 新增配置项

```python
# backend/core/config.py（新增）

# 用户存储配额
```

### Step 2: 新增配额检查服务

```python
# backend/services/quota_service.py（新建）

import os
from pathlib import Path

from core.config import user_max_tasks, user_storage_quota_bytes
from util.file_utils import resolve_user_data_dir

BYTES_PER_MB = 1024 * 1024


class QuotaExceededError(ValueError):
    """配额超限异常。"""
    pass


def _get_user_dir_size(user_data_path: str) -> int:
    """递归计算用户目录总大小（字节）。"""
    user_dir = resolve_user_data_dir(user_data_path)
    if not user_dir.exists():
        return 0

    total = 0
    for entry in user_dir.rglob("*"):
        if entry.is_file():
            try:
                total += entry.stat().st_size
            except OSError:
                pass
    return total


def _get_user_task_count(user_data_path: str) -> int:
    """统计用户任务目录数量。"""
    user_dir = resolve_user_data_dir(user_data_path)
    if not user_dir.exists():
        return 0

    count = 0
    for entry in user_dir.iterdir():
        if entry.is_dir() and entry.name.startswith("task_"):
            count += 1
    return count


def check_upload_quota(user_data_path: str, new_file_size: int) -> None:
    """
    检查上传是否会超出配额。

    Raises:
        QuotaExceededError: 超出配额时抛出。
    """
    current_size = _get_user_dir_size(user_data_path)
    if current_size + new_file_size > user_storage_quota_bytes:
        current_mb = current_size / BYTES_PER_MB
        quota_mb = user_storage_quota_bytes / BYTES_PER_MB
        raise QuotaExceededError(
            f"存储空间不足：已使用 {current_mb:.0f}MB，"
            f"配额 {quota_mb:.0f}MB。请清理旧任务后重试。"
        )

    current_tasks = _get_user_task_count(user_data_path)
    if current_tasks >= user_max_tasks:
        raise QuotaExceededError(
            f"任务数量已达上限（{user_max_tasks}个）。"
            f"请删除旧任务后重试。"
        )


def get_quota_info(user_data_path: str) -> dict:
    """获取用户配额使用情况（供前端展示）。"""
    current_size = _get_user_dir_size(user_data_path)
    current_tasks = _get_user_task_count(user_data_path)

    return {
        "storage_used_bytes": current_size,
        "storage_quota_bytes": user_storage_quota_bytes,
        "storage_used_mb": round(current_size / BYTES_PER_MB, 1),
        "storage_quota_mb": round(user_storage_quota_bytes / BYTES_PER_MB, 1),
        "storage_percent": round(current_size / user_storage_quota_bytes * 100, 1),
        "tasks_used": current_tasks,
        "tasks_quota": user_max_tasks,
    }
```

### Step 4: 添加配额查询 API

```python
# backend/api/quota.py（新建）

from fastapi import APIRouter, Depends
from core.auth import get_current_user
from core.response import success_response
from models.user import User
from services.quota_service import get_quota_info

quota_router = APIRouter(prefix="/api", tags=["配额"])

@quota_router.get("/quota")
async def get_quota(current_user: User = Depends(get_current_user)):
    info = get_quota_info(current_user.data_path)
    return success_response(message="配额信息", data=info)
```

### Step 5: 添加定时清理任务

```python
# backend/services/task_cleanup.py（新建）

import shutil
import time
from pathlib import Path

from core.config import user_task_retention_days
from util.file_utils import user_data_root

RETENTION_SECONDS = user_task_retention_days * 24 * 3600


def cleanup_expired_tasks() -> dict:
    """清理超过保留期的任务目录。返回清理统计。"""
    if not user_data_root.exists():
        return {"deleted": 0, "errors": 0}

    cutoff = time.time() - RETENTION_SECONDS
    deleted = 0
    errors = 0

    for user_dir in user_data_root.iterdir():
        if not user_dir.is_dir():
            continue

        for task_dir in user_dir.iterdir():
            if not task_dir.is_dir() or not task_dir.name.startswith("task_"):
                continue

            try:
                mtime = task_dir.stat().st_mtime
                if mtime < cutoff:
                    shutil.rmtree(task_dir)
                    deleted += 1
            except OSError:
                errors += 1

    return {"deleted": deleted, "errors": errors}
```

在 `backend/main.py` lifespan 中添加定时任务：

```python
async def _periodic_cleanup(interval: int = 3600):
    """每小时执行一次过期任务清理。"""
    from services.task_cleanup import cleanup_expired_tasks

    while True:
        try:
            await asyncio.sleep(interval)
            result = await asyncio.to_thread(cleanup_expired_tasks)
            if result["deleted"] > 0:
                logger.info("清理过期任务: 删除 %d 个", result["deleted"])
        except asyncio.CancelledError:
            break
        except Exception:
            pass
```
