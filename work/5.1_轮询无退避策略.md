# 5.1 轮询无退避策略 — 解决方案

## 问题定位

前端 `ChemSearch.vue` 和 `AdvancedSearch.vue` 中使用了固定 2 秒间隔的轮询，无任何退避策略：

```javascript
// frontend/src/components/ChemSearch.vue:70-73
const RETRIEVE_POLL_INTERVAL_MS = 2000          // 固定 2 秒
const RETRIEVE_POLL_TIMEOUT_MS = 60 * 60 * 1000  // 最长 1 小时
// → 最多 1800 次 HTTP 请求！

const UPLOAD_POLL_INTERVAL_MS = 2000
const UPLOAD_POLL_TIMEOUT_MS = 60 * 60 * 1000
// → 同样 1800 次

// AdvancedSearch.vue:49-50
const RETRIEVE_POLL_INTERVAL_MS = 2000
const RETRIEVE_POLL_TIMEOUT_MS = 10 * 60 * 1000  // 10 分钟
// 注意：与 ChemSearch 的超时时间不一致（1h vs 10min），可能为无心之失
```

轮询循环：

```javascript
// ChemSearch.vue:394-425
const pollUploadResult = async (taskId) => {
  while (!stopUploadPolling) {
    await sleep(UPLOAD_POLL_INTERVAL_MS)  // 总是 sleep 2s
    const statusResp = await getUploadFilesStatus(taskId)
    // ...
  }
}

// ChemSearch.vue:607-646
const pollRetrieveResult = async (jobId) => {
  while (!stopRetrievePolling) {
    await sleep(RETRIEVE_POLL_INTERVAL_MS)  // 总是 sleep 2s
    const statusResp = await getRetrieveStatus(jobId)
    // ...
  }
}
```

## 影响

- 长时间运行的任务产生大量无效轮询请求
- 服务端压力大（尤其是批量任务，30 文件 = 30 job 并发轮询）
- 移动网络下耗电耗流量
- 用户感知差（1 小时的任务最后 30 分钟仍然是 2s 一次轮询）

## 解决方案

### Step 1: 提取通用轮询 hook

```javascript
// frontend/src/composables/usePolling.js（新建）

import { ref, onBeforeUnmount } from 'vue'

/**
 * 带 exponential backoff 的轮询 hook。
 *
 * @param {Function} fetchFn     - 轮询函数，返回 { done: boolean, data: any }
 * @param {Object}   options     - 配置项
 * @param {number}   options.initialInterval - 初始轮询间隔（ms），默认 2000
 * @param {number}   options.maxInterval     - 最大轮询间隔（ms），默认 30000
 * @param {number}   options.backoffFactor   - 退避因子，默认 1.5
 * @param {number}   options.timeout         - 总超时（ms），默认 600000
 * @param {number}   options.fastPhaseMs     - 快速轮询阶段（ms），默认 60000
 */
export function usePolling(fetchFn, options = {}) {
  const {
    initialInterval = 2000,
    maxInterval = 30000,
    backoffFactor = 1.5,
    timeout = 600000,
    fastPhaseMs = 60000,
  } = options

  const isPolling = ref(false)
  const error = ref(null)
  let abortController = null
  let timerId = null

  /**
   * 计算下一次轮询间隔：
   * - 前 fastPhaseMs 毫秒内：固定 initialInterval
   * - 之后：exponential backoff，上限 maxInterval
   */
  function calcInterval(elapsedMs) {
    if (elapsedMs < fastPhaseMs) {
      return initialInterval
    }
    // 超过快速阶段后的退避次数
    const backoffSteps = Math.floor((elapsedMs - fastPhaseMs) / maxInterval) + 1
    const interval = Math.min(
      initialInterval * Math.pow(backoffFactor, backoffSteps),
      maxInterval,
    )
    return Math.round(interval)
  }

  async function start() {
    if (isPolling.value) return
    isPolling.value = true
    error.value = null
    abortController = new AbortController()

    const startedAt = Date.now()

    while (isPolling.value) {
      const elapsed = Date.now() - startedAt

      // 超时检查
      if (elapsed >= timeout) {
        error.value = new Error('请求超时，请重试')
        isPolling.value = false
        return
      }

      try {
        const result = await fetchFn(abortController.signal)

        if (result.done) {
          isPolling.value = false
          return result.data
        }
      } catch (err) {
        if (err.name === 'AbortError' || err.name === 'CanceledError') {
          isPolling.value = false
          return
        }
        // 网络错误不立即中止，等下一次轮询重试
        error.value = err
      }

      // 计算下次间隔
      const nextInterval = calcInterval(Date.now() - startedAt)

      // 等待下一次轮询
      await new Promise((resolve) => {
        timerId = setTimeout(resolve, nextInterval)
      })
    }
  }

  function stop() {
    isPolling.value = false
    if (abortController) {
      abortController.abort()
      abortController = null
    }
    if (timerId) {
      clearTimeout(timerId)
      timerId = null
    }
  }

  onBeforeUnmount(() => {
    stop()
  })

  return { isPolling, error, start, stop }
}
```

```javascript

  async (signal) => {

      signal,  // 传递 AbortSignal 给 axios

      error: statusData.error || null,

    return { done: false }
  },
  {
    initialInterval: 2000,
    maxInterval: 15000,     // 最多 15 秒轮询一次
```

### Step 3: 同样改造上传轮询

```javascript
// frontend/src/components/ChemSearch.vue

const {
  isPolling: isUploadPolling,
  error: uploadPollError,
  start: startUploadPoll,
  stop: stopUploadPoll,
} = usePolling(
  async (signal) => {
    const statusResp = await getUploadFilesStatus(paths.taskId, { signal })
    const statusData = statusResp?.data?.data
    if (!statusData) throw new Error('Failed to read upload task status')

    const progress = statusData.progress || {}
    if (progress.total) {
      uploadProgress.value = Math.round(((progress.done || 0) / progress.total) * 100)
    }

    if (['success', 'partial_failed'].includes(statusData.status)) {
      // ...保存状态...
      return { done: true, data: statusData }
    }

    if (statusData.status === 'failed') {
      // ...
      return { done: true, data: null }
    }

    return { done: false }
  },
  {
    initialInterval: 2000,
    maxInterval: 10000,
    timeout: 60 * 60 * 1000,
```

| 0-1 min | 2s | 30 | 2s | 30 |

| 3-5 min | 2s | 150 | 4.5s | ~64 |
| 5-10 min | 2s | 300 | 6.75s | ~86 |
| 10-30 min | 2s | 900 | 15s | ~166 |
